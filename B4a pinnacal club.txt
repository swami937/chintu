#include <iostream>
#include <string>
using namespace std;

// Node structure
struct Node {
    string name;
    long long prn;
    Node* next;
};

// Head pointer (President)
Node* head = NULL;

// Function to add President
void addPresident(string name, long long prn) {
    Node* newNode = new Node();
    newNode->name = name;
    newNode->prn = prn;
    newNode->next = head;
    head = newNode;
    cout << "President added successfully.\n";
}

// Function to add Secretary
void addSecretary(string name, long long prn) {
    Node* newNode = new Node();
    newNode->name = name;
    newNode->prn = prn;
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
    } else {
        Node* temp = head;
        while (temp->next != NULL)
            temp = temp->next;
        temp->next = newNode;
    }
    cout << "Secretary added successfully.\n";
}

// Function to add Member (between president and secretary)
void addMember(string name, long long prn) {
    if (head == NULL) {
        cout << "Add president first!\n";
        return;
    }

    Node* newNode = new Node();
    newNode->name = name;
    newNode->prn = prn;

    Node* temp = head;
    while (temp->next != NULL && temp->next->next != NULL)
        temp = temp->next;

    newNode->next = temp->next;
    temp->next = newNode;
    cout << "Member added successfully.\n";
}

// Function to display all members
void display() {
    Node* temp = head;
    if (temp == NULL) {
        cout << "No members in the club.\n";
        return;
    }
    cout << "\n--- Club Members ---\n";
    while (temp != NULL) {
        cout << "PRN: " << temp->prn << " | Name: " << temp->name << endl;
        temp = temp->next;
    }
}

// Function to count total members
void countMembers() {
    Node* temp = head;
    int count = 0;
    while (temp != NULL) {
        count++;
        temp = temp->next;
    }
    cout << "\nTotal Members: " << count << endl;
}

// Main Function
int main() {
    int choice;
    string name;
    long long prn;

    do {
        cout << "\n===== Pinnacle Club Menu =====";
        cout << "\n1. Add President";
        cout << "\n2. Add Member";
        cout << "\n3. Add Secretary";
        cout << "\n4. Display Members";
        cout << "\n5. Count Members";
        cout << "\n6. Exit";
        cout << "\nEnter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter President Name: ";
                cin >> name;
                cout << "Enter PRN: ";
                cin >> prn;
                addPresident(name, prn);
                break;

            case 2:
                cout << "Enter Member Name: ";
                cin >> name;
                cout << "Enter PRN: ";
                cin >> prn;
                addMember(name, prn);
                break;

            case 3:
                cout << "Enter Secretary Name: ";
                cin >> name;
                cout << "Enter PRN: ";
                cin >> prn;
                addSecretary(name, prn);
                break;

            case 4:
                display();
                break;

            case 5:
                countMembers();
                break;

            case 6:
                cout << "Exiting program.\n";
                break;

            default:
                cout << "Invalid choice.\n";
        }
    } while (choice != 6);

    return 0;
}
