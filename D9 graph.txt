#include <iostream>
using namespace std;

#define MAX 10

class Graph {
    int n;                // number of cities
    int adj[MAX][MAX];    // adjacency matrix
    int visited[MAX];

public:
    Graph() {
        n = 0;
        for (int i = 0; i < MAX; i++)
            for (int j = 0; j < MAX; j++)
                adj[i][j] = 0;
    }

    // Create graph
    void create() {
        cout << "Enter number of cities: ";
        cin >> n;

        cout << "\nEnter adjacency matrix (0 = no flight, cost = flight time/fuel):\n";
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cin >> adj[i][j];
            }
        }
    }

    // Display adjacency matrix
    void display() {
        cout << "\nAdjacency Matrix:\n";
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++)
                cout << adj[i][j] << " ";
            cout << endl;
        }
    }

    // DFS traversal
    void DFS(int v) {
        visited[v] = 1;
        for (int i = 0; i < n; i++) {
            if (adj[v][i] != 0 && !visited[i])
                DFS(i);
        }
    }

    // Check if graph is connected
    void checkConnected() {
        for (int i = 0; i < n; i++)
            visited[i] = 0;

        DFS(0);  // start DFS from first city

        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                cout << "\nGraph is NOT connected.\n";
                return;
            }
        }
        cout << "\nGraph is CONNECTED.\n";
    }
};

int main() {
    Graph g;
    g.create();
    g.display();
    g.checkConnected();
    return 0;
}
