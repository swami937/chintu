#include <iostream>
using namespace std;

int linearSearch(int arr[], int n, int key) {
    for (int i = 0; i < n; i++)
        if (arr[i] == key)
            return i;
    return -1;
}

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (arr[j] > arr[j + 1])
                swap(arr[j], arr[j + 1]);
}

int main() {
    int roll[50], n, ch, key;
    cout << "Enter number of students: ";
    cin >> n;
    cout << "Enter roll numbers:\n";
    for (int i = 0; i < n; i++) cin >> roll[i];

    do {
        cout << "\n--- MENU ---\n";
        cout << "1. Sort Roll Numbers\n2. Search Roll Number\n3. Display List\n4. Exit\n";
        cout << "Enter your choice: ";
        cin >> ch;

        switch (ch) {
            case 1:
                bubbleSort(roll, n);
                cout << "Roll numbers sorted.\n";
                break;
            case 2:
                cout << "Enter roll number to search: ";
                cin >> key;
                if (linearSearch(roll, n, key) != -1)
                    cout << "Student attended training.\n";
                else
                    cout << "Student did NOT attend.\n";
                break;
            case 3:
                cout << "Roll numbers: ";
                for (int i = 0; i < n; i++) cout << roll[i] << " ";
                cout << endl;
                break;
            case 4:
                cout << "Exiting...\n";
                break;
            default:
                cout << "Invalid choice.\n";
        }
    } while (ch != 4);

    return 0;
}
