#include <iostream>
using namespace std;

// ---------- Binary Search Function ----------
int binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1, mid;
    while (low <= high) {
        mid = (low + high) / 2;
        if (arr[mid] == key)
            return mid;        // found
        else if (arr[mid] < key)
            low = mid + 1;     // search in right half
        else
            high = mid - 1;    // search in left half
    }
    return -1; // not found
}

// ---------- Bubble Sort Function ----------
void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    cout << "Roll numbers sorted successfully.\n";
}

// ---------- Display Function ----------
void display(int arr[], int n) {
    cout << "Roll numbers: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl;
}

// ---------- Main Function ----------
int main() {
    int roll[50], n, ch, key;

    cout << "Enter number of students: ";
    cin >> n;
    cout << "Enter roll numbers:\n";
    for (int i = 0; i < n; i++)
        cin >> roll[i];

    do {
        cout << "\n--- MENU ---\n";
        cout << "1. Sort Roll Numbers\n";
        cout << "2. Search Roll Number (Binary Search)\n";
        cout << "3. Display Roll Numbers\n";
        cout << "4. Exit\n";
        cout << "Enter your choice: ";
        cin >> ch;

        switch (ch) {
            case 1:
                bubbleSort(roll, n);
                break;

            case 2:
                cout << "Enter roll number to search: ";
                cin >> key;
                int pos;
                pos = binarySearch(roll, n, key);
                if (pos != -1)
                    cout << "Student with roll number " << key << " attended the training program.\n";
                else
                    cout << "Student with roll number " << key << " did NOT attend.\n";
                break;

            case 3:
                display(roll, n);
                break;

            case 4:
                cout << "Exiting program...\n";
                break;

            default:
                cout << "Invalid choice. Try again.\n";
        }

    } while (ch != 4);

    return 0;
}
