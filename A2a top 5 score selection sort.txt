#include <iostream>
using namespace std;


void SelectionSort(float arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIndex = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex])
                minIndex = j;
        }
       
        float temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }
    cout << "Array sorted successfully (ascending order)." << endl;
}


void DisplayTop5(float arr[], int n) {
    cout << "Top 5 Scores: ";
    int count = 0;
    // Array is sorted ascending ? so start from end
    for (int i = n - 1; i >= 0 && count < 5; i--) {
        cout << arr[i] << " ";
        count++;
    }
    cout << endl;
}


void Display(float arr[], int n) {
    cout << "Student Percentages: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl;
}


int main() {
    int n, ch;
    float perc[50];

    cout << "Enter number of students: ";
    cin >> n;

    cout << "Enter first year percentages:\n";
    for (int i = 0; i < n; i++)
        cin >> perc[i];

    do {
        cout << "\n--- MENU ---\n";
        cout << "1. Sort Percentages (Selection Sort)\n";
        cout << "2. Display All Percentages\n";
        cout << "3. Display Top 5 Scores\n";
        cout << "4. Exit\n";
        cout << "Enter your choice: ";
        cin >> ch;

        switch (ch) {
            case 1:
                SelectionSort(perc, n);
                break;

            case 2:
                Display(perc, n);
                break;

            case 3:
                SelectionSort(perc, n); // Ensure sorted before showing top 5
                DisplayTop5(perc, n);
                break;

            case 4:
                cout << "Exiting program..." << endl;
                break;

            default:
                cout << "Invalid choice, please try again." << endl;
        }
    } while (ch != 4);

    return 0;
}





Program Logic (Short & Attractive)

The program stores students’ percentages and provides a menu to sort or display them.

Selection Sort is used to arrange percentages in ascending order by repeatedly finding the minimum element and swapping it to the front.

Display() function prints all percentages.

DisplayTop5() shows the top 5 highest percentages by printing from the end of the sorted list.

A menu-driven loop allows the user to:

Sort data

Display all percentages

Display top 5 scorers

Exit the program

⚙️ Selection Sort Logic (Short)

For each position i, find the smallest element in the remaining array.

Swap it with the element at position i.

Repeat until the whole array is sorted.

⏱ Time Complexity

Best / Worst / Average Case: O(n²)

Works well for small datasets.
